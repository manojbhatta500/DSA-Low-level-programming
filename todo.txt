fn load_files_recursive(path: &Path, files: &mut Vec<String>) {
    if path.is_dir() {
        for entry in fs::read_dir(path).unwrap() {
            let entry = entry.unwrap();
            let path = entry.path();
            if path.is_file() {
                files.push(path.to_str().unwrap().to_string());
            } else if path.is_dir() {
                load_files_recursive(&path, files); // recursive call
            }
        }
    }
}
